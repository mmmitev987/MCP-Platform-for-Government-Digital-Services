# ── Core MCP framework ──────────────────────────────────────────────────────
# FastMCP is the high-level Python wrapper around the official MCP SDK.
# Used by institution servers (institutions/*/main.py).
# The gateway (gateway/main.py) uses the low-level mcp.server.lowlevel.Server.
mcp[cli]

# ── Browser automation (primary login strategy) ──────────────────────────────
# Playwright launches a real Chromium browser so the user can log in manually
# through any dynamic/popup-based auth flow (SSO, CAPTCHA, eID, etc.)
playwright

# ── HTTP client (tool requests and http-based auth fallback) ─────────────────
requests

# ── Google Gemini SDK (new "google-genai" package, NOT google-generativeai) ──
google-genai

# ── Environment variable loading (.env file support) ─────────────────────────
python-dotenv

# ── Symmetric encryption for cookie persistence ──────────────────────────────
# Fernet (part of the cryptography package) encrypts saved cookies so they
# are never stored in plaintext on disk.
cryptography

# ── YAML parser — used by the gateway to read gateway/config.yaml ────────────
# PyYAML is the standard Python YAML library.
PyYAML
